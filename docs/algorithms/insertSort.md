# 插入排序

利用双指针

```js
/**
 *     i -> 往前遍历，从1开始遍历，最后一位无需遍历
 * [11,3,14,54,67,332,12,3]
 *  <- j 往后遍历，拿i位置的值与前面的值逐一进行比较
 */

var arr = [11,3,14,54,67,332,12,3]

function insertSort(arr){
  // 定义指针i，从数组第二个元素开始遍历
  for(var i=1;i<arr.length;i++){
    // 定义一变量temp，保存指针i对应的元素值
    var temp = arr[i]
    // 定义指针j，从指针i所在位置反向遍历
    for(var j = i; j>0; j--){
      // 拿temp与前面的元素逐个对比，如果大于或等于前面元素，说明无需排序，直接跳出遍历。否则把上一个元素的值赋值给当前指针j的元素，j--，进入下一轮遍历
      if(temp >= arr[j-1]){
        break
      }
      arr[j] = arr[j-1]
    }
    /**
     * 指针j遍历结束后，j要么等于0，要么等于跳出循环时的值，
     * 如果j等于0，说明temp为最小值，所以要把temp赋值到0的位置
     * 如果是跳出循环时的值，说明temp大于或等于j-1位置的值，小于j+1位置的值，还要把temp赋值到j的位置
     */
    arr[j] = temp
  }
  return arr
}

var res = insertSort(arr)

console.log(res) // => [3, 3, 11, 12, 14, 54, 67, 332]
```
